# using Singular Value Decomposition (SVD)
def low_rank(X, M=1):
    U, D, V = np.linalg.svd(X, full_matrices=False)
    L = U[:,:M] * D[None,:M]
    return L.dot(V[:M])
In[99]:
# Replace missing values in X_scaled with 0 
ismiss = np.isnan(X_scaled)
In[100]:
Xhat = X_scaled.copy()
Xhat[ismiss] = 0
In[101]:
# Parameters for the iterative matrix completion algorithm
thresh = 1e-5     
rel_err = 1
count = 0
mssold = np.mean(Xhat[~ismiss]**2)
mss0 = np.mean(X_scaled[~ismiss]**2)
n_components = 5
In[102]:
# Iterative SVD Matrix Completion
print("Starting iterative imputation.")

while rel_err > thresh:
    count += 1
     # Step 2(a): Low-rank approximation using SVD (PCA)
    Xapp = low_rank(Xhat, M=n_components)
    
    # Step 2(b): Update ONLY the missing values
    Xhat[ismiss] = Xapp[ismiss]
    # Step 2(c): Compute relative error on observed (non-missing) entries
    mss = np.mean(((X_scaled - Xapp)[~ismiss])**2)
    rel_err = (mssold - mss) / mss0
    mssold = mss
    
    print(f"Iteration: {count}, Rel.Err: {rel_err:.2e}")

print("Convergence reached.")

Interpretation

from sklearn.ensemble import GradientBoostingRegressor
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error
from sklearn.utils import resample
from sklearn.preprocessing import StandardScaler
X = train[X_cols] 
Y = train['AQI']

data_full = pd.concat([X, Y], axis=1)
GBR_PARAMS = {
    'n_estimators': 200, 
    'learning_rate': 0.05, 
    'max_depth': 5,
    'random_state': 42 
}
N_BOOTSTRAPS = 100 
MSE_BOOTSTRAP_RESULTS = []
In[139]:
print(f" Launching {N_BOOTSTRAPS} Bootstrapping iterations")

for i in range(N_BOOTSTRAPS):
    train_bootstrapped = resample(data_full, replace=True, random_state=i)
    
    
    X_boot_train, X_boot_val, Y_boot_train, Y_boot_val = train_test_split(
        train_bootstrapped[X_cols], 
        train_bootstrapped['AQI'], 
        test_size=0.2, 
        random_state=i 
    )

    scaler = StandardScaler()
    X_boot_train_scaled = scaler.fit_transform(X_boot_train)
    X_boot_val_scaled = scaler.transform(X_boot_val)

    gbr_model = GradientBoostingRegressor(**GBR_PARAMS)
    gbr_model.fit(X_boot_train_scaled, Y_boot_train)

    Y_pred_boot_val = gbr_model.predict(X_boot_val_scaled)
    mse_i = mean_squared_error(Y_boot_val, Y_pred_boot_val)
    
    MSE_BOOTSTRAP_RESULTS.append(mse_i)
    
    if (i+1) % 20 == 0:
        print(f"  Iteration {i+1}/{N_BOOTSTRAPS} finish. MSE : {mse_i:.3f}")

Conclusion

The performance comparison for AQI regression and pollutant classification shows that non-linear tree-based methods are clearly superior. Linear models (like Lasso and Ridge) reach their limit with an RMSE around 8.4 for AQI, confirming that the pollutantAQI relationship is strongly non-linear.

Our final chosen model is Gradient Boosting. For AQI regression, it achieves an RMSE of 4.920 (with a 95% confidence interval for the True Test MSE of [21.274, 28.126]). For pollutant classification, it reaches an Accuracy of 0.9687. These results show that predictions are mainly driven by Fine Particles (PM2.5_AVG), which represent more than 90% of the predictive importance.

Looking ahead, adding clustering concepts (Smog, Traffic, Dust) and exploring a Deep Learning model such as an LSTM for short-term forecasting are promising next steps to turn these results into real environmental decision-making tools.
